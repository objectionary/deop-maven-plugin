24  package org.eolang.opeo.ast;

26  import com.jcabi.matchers.XhtmlMatchers;
27  import java.util.stream.Collectors;
28  import java.util.stream.Stream;
29  import org.eolang.jeo.representation.HexData;
30  import org.eolang.jeo.representation.xmir.XmlNode;
31  import org.eolang.opeo.compilation.HasInstructions;
32  import org.hamcrest.MatcherAssert;
33  import org.hamcrest.Matchers;
34  import org.junit.jupiter.api.Test;
35  import org.junit.jupiter.params.ParameterizedTest;
36  import org.junit.jupiter.params.provider.Arguments;
37  import org.junit.jupiter.params.provider.MethodSource;
38  import org.objectweb.asm.Opcodes;
39  import org.objectweb.asm.Type;
40  import org.xembly.ImpossibleModificationException;
41  import org.xembly.Xembler;
42  
43  /**
44   * Test for {@link LocalVariable}.
45   * @since 0.2
46   */
47  final class LocalVariableTest {
48  
49      @Test
50      void convertsToXmir() throws ImpossibleModificationException {
51          MatcherAssert.assertThat(
52              "We expect the xmir variable to be equal to <o base='local1'/>, but it wasn't",
53              new Xembler(new LocalVariable(1, Type.INT_TYPE).toXmir()).xml(),
54              XhtmlMatchers.hasXPath("./o[@base='local-1']")
55          );
56      }
57  
58      @ParameterizedTest(name = "[{index}] {0} => {1}")
59      @MethodSource("typesArguments")
60      void convertsType(
61          final Type type, final String expected
62      ) throws ImpossibleModificationException {
63          final String xml = new Xembler(new LocalVariable(1, type).toXmir()).xml();
64          MatcherAssert.assertThat(
65              String.format(
66                  "We expect the xmir variable type to be equal to <o base='local1' scope='%s'/>, but it wasn't, current value is '%s'",
67                  expected,
68                  xml
69              ),
70              xml,
71              Matchers.containsString(new HexData(expected).value())
72          );
73      }
74  
75      @Test
76      void createsVariableFromXmir() throws ImpossibleModificationException {
77          final LocalVariable original = new LocalVariable(2, Type.FLOAT_TYPE);
78          MatcherAssert.assertThat(
79              "Can't correctly create variable from XMIR. We expect the variable to be equal to the original, but it wasn't",
80              new LocalVariable(new XmlNode(new Xembler(original.toXmir()).xml())),
81              Matchers.equalTo(original)
82          );
83      }
84  
85      @ParameterizedTest(name = "[{index}] {0} => {1}")
86      @MethodSource("loads")
87      void transformsToBytecodeInstructions(final Type type, final int expected) {
88          MatcherAssert.assertThat(
89              "Can't correctly transform variable to bytecode instructions. It should be exactly 1 instruction",
90              new LocalVariable(0, type).opcodes().stream().map(AstNode::toXmir)
91                  .map(Xembler::new)
92                  .map(Xembler::xmlQuietly)
93                  .map(XmlNode::new)
94                  .collect(Collectors.toList()),
95              new HasInstructions(expected)
96          );
97      }
98  
99      /**
100      * Types arguments.
101      * Don't remove this method, it's used by {@link #convertsType(Type, String)}.
102      * @return Arguments.
103      * @checkstyle UnusedPrivateMethod (10 lines)
104      */
105     @SuppressWarnings("PMD.UnusedPrivateMethod")
106     private static Stream<Arguments> typesArguments() {
107         return Stream.of(
108             Arguments.of(Type.INT_TYPE, "I"),
109             Arguments.of(Type.BOOLEAN_TYPE, "Z"),
110             Arguments.of(Type.BYTE_TYPE, "B"),
111             Arguments.of(Type.CHAR_TYPE, "C"),
112             Arguments.of(Type.DOUBLE_TYPE, "D"),
113             Arguments.of(Type.FLOAT_TYPE, "F"),
114             Arguments.of(Type.LONG_TYPE, "J"),
115             Arguments.of(Type.SHORT_TYPE, "S"),
116             Arguments.of(Type.VOID_TYPE, "V"),
117             Arguments.of(Type.getType("Ljava/lang/String;"), "Ljava/lang/String;")
118         );
119     }
120 
121     /**
122      * Arguments for {@link #transformsToBytecodeInstructions(Type, int)} ()}.
123      * Don't remove this method, it's used by {@link #transformsToBytecodeInstructions(Type, int)}}.
124      * @return Arguments.
125      */
126     @SuppressWarnings("PMD.UnusedPrivateMethod")
127     private static Stream<Arguments> loads() {
128         return Stream.of(
129             Arguments.of(Type.INT_TYPE, Opcodes.ILOAD),
130             Arguments.of(Type.BOOLEAN_TYPE, Opcodes.ILOAD),
131             Arguments.of(Type.BYTE_TYPE, Opcodes.ILOAD),
132             Arguments.of(Type.CHAR_TYPE, Opcodes.ILOAD),
133             Arguments.of(Type.DOUBLE_TYPE, Opcodes.DLOAD),
134             Arguments.of(Type.FLOAT_TYPE, Opcodes.FLOAD),
135             Arguments.of(Type.LONG_TYPE, Opcodes.LLOAD),
136             Arguments.of(Type.SHORT_TYPE, Opcodes.ILOAD),
137             Arguments.of(Type.getType("Ljava/lang/String;"), Opcodes.ALOAD)
138         );
139     }

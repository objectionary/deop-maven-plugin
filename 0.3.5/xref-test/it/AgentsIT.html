53  /**
54   * Test pack that checks how separate agents decompile instructions.
55   * @since 0.4
56   * @todo #381:30min Remove `program` and `object` elements.
57   *  We added this elements to the test to make it pass.
58   *  Currently {@link Xmir.Default} does not support the decompilation of partial XMIR documents.
59   *  So, by adding these elements we mimic the full XMIR document.
60   *  When the related issue is fixed, remove these elements.
61   *  You can follow the progress of the issue
62   *  <a href="https://github.com/objectionary/eo/issues/3343">here.</a>
63   * @todo #404:90min Add More Tests.
64   *  Currently, we have only the small number of test packs that checks instructions decompilation.
65   *  We need to add more complicated tests that involve more agents.
66   */
67  @SuppressWarnings("JTCOP.RuleCorrectTestName")
68  final class AgentsIT {
69  
70      @ParameterizedTest
71      @ClasspathSource(value = "agents", glob = "**.yaml")
72      void verifiesAgents(final String yaml) {
73          final InstructionPack pack = new InstructionPack(yaml);
74          final TracedAgent.Container output = new TracedAgent.Container();
75          MatcherAssert.assertThat(
76              "Agents should decompile instructions correctly, according to the YAML pack",
77              new Xmir.Default(
78                  new XMLDocument(
79                      new Xembler(
80                          new Directives().add("program").add("objects")
81                              .append(
82                                  new DecompilerMachine(Collections.singletonMap("output", output))
83                                      .decompile(pack.instructions())
84                              )
85                              .up().up()
86                      ).xmlQuietly()
87                  )
88              ).toEO(),
89              Matchers.equalTo(pack.expected())
90          );
91          MatcherAssert.assertThat(
92              "We expect that agents is used in the same order as expected, but they didn't",
93              output.agentsUsed(),
94              Matchers.equalTo(pack.agents())
95          );
96      }

98      /**
99       * Instruction pack.
100      * You can find examples of the {@link InstructionPack} right
101      * <a href="./resources/agents">here.</a>
102      * @since 0.4
103      */
104     private static final class InstructionPack {
105 
106         /**
107          * The pattern to match instructions.
108          * It matches:
109          * 1. Boolean values. Example: true, false.
110          * 2. Double values. Example: 10.12.
111          * 3. Long values. Example: 100L.
112          * 4. Integer values. Example: 100.
113          * 5. String values. Example: "Hello world!".
114          * 6. Label. Example: LABEL:labelid.
115          * 7. Instruction names. Example: LDC.
116          */
117         private static final Pattern INSTRUCTION = Pattern.compile(
118             "(\\bfalse|true\\b)|(\\d+\\.\\d+)|(\\d+L)|(\\d+)|\"([^\"]*)\"|\\bLABEL\\b:(\\S+)|(\\S+)"
119         );
120 
121         /**
122          * Yaml pack.
123          */
124         private final Scalar<? extends Map<String, Object>> pack;
125 
126         /**
127          * Constructor.
128          * @param yaml Yaml file content.
129          */
130         InstructionPack(final String yaml) {
131             this(new Sticky<>(() -> new Yaml().load(yaml)));
132         }
133 
134         /**
135          * Constructor.
136          * @param pack Yaml pack.
137          */
138         private InstructionPack(final Scalar<? extends Map<String, Object>> pack) {
139             this.pack = pack;
140         }
141 
142         /**
143          * Expected agents used to decompile instructions.
144          * @return Agents.
145          */
146         @SuppressWarnings("unchecked")
147         List<String> agents() {
148             return (List<String>) this.value("agents");
149         }
150 
151         /**
152          * Expected EO code.
153          * @return EO code.
154          */
155         String expected() {
156             return (String) this.value("eo");
157         }
158 
159         /**
160          * Instructions to decompile.
161          * @return Instructions.
162          */
163         @SuppressWarnings("unchecked")
164         Instruction[] instructions() {
165             return ((Collection<String>) this.value("opcodes"))
166                 .stream()
167                 .map(this::parse)
168                 .toArray(Instruction[]::new);
169         }
170 
171         /**
172          * Parse instruction.
173          * @param instr Instruction as string.
174          * @return Instruction.
175          */
176         private Instruction parse(final String instr) {
177             final Matcher matcher = InstructionPack.INSTRUCTION.matcher(instr);
178             final AtomicInteger opcode = new AtomicInteger();
179             final List<Object> arguments = new ArrayList<>(0);
180             while (matcher.find()) {
181                 if (Objects.nonNull(matcher.group(1))) {
182                     arguments.add(Boolean.parseBoolean(matcher.group(1)));
183                 } else if (Objects.nonNull(matcher.group(2))) {
184                     arguments.add(Double.parseDouble(matcher.group(2)));
185                 } else if (Objects.nonNull(matcher.group(3))) {
186                     arguments.add(Long.parseLong(matcher.group(3)));
187                 } else if (Objects.nonNull(matcher.group(4))) {
188                     arguments.add(Integer.parseInt(matcher.group(4)));
189                 } else if (Objects.nonNull(matcher.group(5))) {
190                     final String group = matcher.group(5);
191                     arguments.add(group);
192                 } else if (Objects.nonNull(matcher.group(6))) {
193                     arguments.add(new AllLabels().label(matcher.group(6)));
194                 } else {
195                     opcode.set(new OpcodeName(matcher.group(7)).code());
196                 }
197             }
198             return new OpcodeInstruction(opcode.get(), arguments.toArray());
199         }
200 
201         /**
202          * Get Yaml value by key.
203          * @param key Yaml key.
204          * @return Yaml value.
205          * @checkstyle IllegalCatchCheck (6 lines)
206          */
207         @SuppressWarnings("PMD.AvoidCatchingGenericException")
208         private Object value(final String key) {
209             try {
210                 return this.pack.value().get(key);
211             } catch (final Exception exception) {
212                 throw new IllegalStateException("Failed to parse YAML pack", exception);
213             }
214         }
215     }

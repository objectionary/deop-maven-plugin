24  package org.eolang.opeo.decompilation;
25  
26  import java.util.ArrayDeque;
27  import java.util.Deque;
28  import java.util.Iterator;
29  import java.util.LinkedList;
30  import java.util.List;
31  import java.util.Optional;
32  import lombok.ToString;
33  import org.eolang.opeo.ast.AstNode;
34  import org.eolang.opeo.ast.Label;
35  import org.eolang.opeo.ast.Labeled;
36  
37  /**
38   * Operand stack.
39   * <p>
40   * You can read more about it
41   * <a href="https://stackoverflow.com/questions/24427056/what-is-an-operand-stack">here</a>
42   * </p>
43   * It emulates the behaviour of JVM operand stack.
44   * @since 0.2
45   */
46  @ToString
47  public final class OperandStack {
48  
49      /**
50       * Output Stack.
51       */
52      private final Deque<AstNode> stack;
53  
54      /**
55       * Default constructor.
56       */
57      OperandStack() {
58          this(new ArrayDeque<>(0));
59      }
60  
61      /**
62       * Constructor.
63       * @param original Initial stack collection.
64       */
65      private OperandStack(final Deque<AstNode> original) {
66          this.stack = original;
67      }
68  
69      /**
70       * Pop one node from the stack or return empty.
71       * @return Optional node.
72       */
73      public Optional<AstNode> first() {
74          final Optional<AstNode> result;
75          if (this.stack.isEmpty()) {
76              result = Optional.empty();
77          } else {
78              result = Optional.ofNullable(this.stack.pop());
79          }
80          return result;
81      }
82  
83      /**
84       * Pop one node from the stack.
85       * @return Node.
86       */
87      public AstNode pop() {
88          final AstNode res;
89          final AstNode pop = this.stack.pop();
90          if (pop instanceof Label) {
91              res = new Labeled(this.stack.pop(), (Label) pop);
92          } else {
93              res = pop;
94          }
95          return res;
96      }
97  
98      /**
99       * Pop N nodes from the stack.
100      * @param number Number of nodes to pop.
101      * @return Collection of nodes.
102      */
103     public List<AstNode> pop(final int number) {
104         final List<AstNode> args = new LinkedList<>();
105         for (int index = 0; index < number; ++index) {
106             args.add(this.pop());
107         }
108         return args;
109     }
110 
111     /**
112      * Push one more node to the stack.
113      * @param node Node to add to the stack.
114      */
115     public void push(final AstNode node) {
116         this.stack.push(node);
117     }
118 
119     /**
120      * Duplicate the higher value on the stack.
121      */
122     public void dup() {
123         this.stack.push(this.stack.peek());
124     }
125 
126     /**
127      * Retrieve iterator of the stack in the reverse order.
128      * @return Nodes iterator.
129      */
130     Iterator<AstNode> descendingIterator() {
131         return this.stack.descendingIterator();
132     }
133 }

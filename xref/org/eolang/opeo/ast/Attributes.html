40  /**
41   * Type attributes of AST nodes.
42   * This class is useful when we need to preserve the information about types of AST nodes.
43   * @since 0.1
44   */
45  @SuppressWarnings("PMD.TooManyMethods")
46  @EqualsAndHashCode
47  public final class Attributes implements Xmir {
48  
49      /**
50       * New line pattern for replacing redundant symbols from hex strings.
51       */
52      private static final Pattern NEW_LINE = Pattern.compile("\n", Pattern.LITERAL);
53  
54      /**
55       * All attributes.
56       */
57      private final Map<String, String> all;
58  
59      /**
60       * Constructor.
61       * @param raw Raw attributes
62       */
63      public Attributes(final String raw) {
64          this(Attributes.parse(raw));
65      }
66  
67      /**
68       * Constructor.
69       * @param entries Attribute entry pairs.
70       */
71      public Attributes(final String... entries) {
72          this(Attributes.fromEntries(entries));
73      }
74  
75      /**
76       * Constructor.
77       * @param node Xmir representation of attributes.
78       */
79      public Attributes(final XmlNode node) {
80          this(Attributes.fromXmir(node));
81      }
82  
83      /**
84       * Constructor.
85       * @param all All attributes.
86       */
87      public Attributes(final Map<String, String> all) {
88          this.all = new TreeMap<>(all);
89      }
90  
91      @Override
92      public Directives toXmir() {
93          return new Directives(new DirectivesData(this.toString()));
94      }
95  
96      @Override
97      public String toString() {
98          return this.all.entrySet().stream()
99              .map(entry -> String.format("%s=%s", entry.getKey(), entry.getValue()))
100             .collect(Collectors.joining("|"));
101     }
102 
103     /**
104      * Get descriptor attribute.
105      * @return Descriptor value.
106      */
107     public String descriptor() {
108         return this.find("descriptor");
109     }
110 
111     /**
112      * Set descriptor attribute.
113      * @param descriptor Descriptor
114      * @return This object
115      */
116     public Attributes descriptor(final String descriptor) {
117         this.all.put("descriptor", descriptor);
118         return this;
119     }
120 
121     /**
122      * Get type attribute.
123      * @return Type value.
124      */
125     public String type() {
126         return this.find("type");
127     }
128 
129     /**
130      * Set type attribute.
131      * @param type Type
132      * @return This object
133      */
134     public Attributes type(final String type) {
135         this.all.put("type", type);
136         return this;
137     }
138 
139     /**
140      * Get owner attribute.
141      * @return Owner value.
142      */
143     public String owner() {
144         return this.find("owner");
145     }
146 
147     /**
148      * Set owner attribute.
149      * @param owner Owner
150      * @return This object
151      */
152     public Attributes owner(final String owner) {
153         this.all.put("owner", owner);
154         return this;
155     }
156 
157     /**
158      * Get name attribute.
159      * @return Name value.
160      */
161     public String name() {
162         return this.find("name");
163     }
164 
165     /**
166      * Set name attribute.
167      * @param name Name
168      * @return This object
169      */
170     public Attributes name(final String name) {
171         this.all.put("name", name);
172         return this;
173     }
174 
175     /**
176      * Get interfaced attribute.
177      * @return True if method is interfaced or not.
178      * @todo #201:90min Fix 'staticize' optimization.
179      *  We need to use of the default value below because outdated 'staticize' optimization
180      *  doesn't add 'interfaced' attribute to some methods. Particulary,
181      *  scope="name=get|descriptor=()I|owner=org/eolang/benchmark/StaticizedA|type=method"
182      *  in class A should contain 'interfaced=false'.
183      *  To implement this, we need to send a PR to ineo-maven-plugin repository.
184      *  See https://github.com/objectionary/ineo-maven-plugin
185      */
186     public boolean interfaced() {
187         return "true".equals(this.all.getOrDefault("interfaced", "false"));
188     }
189 
190     /**
191      * Set interfaced attribute.
192      * @param interfaced Interfaced method or not
193      * @return This object
194      */
195     public Attributes interfaced(final boolean interfaced) {
196         this.all.put("interfaced", Boolean.toString(interfaced));
197         return this;
198     }
199 
200     /**
201      * Find attribute.
202      * @param key Attribute key
203      * @return Attribute value
204      */
205     private String find(final String key) {
206         if (this.all.containsKey(key)) {
207             return this.all.get(key);
208         } else {
209             throw new IllegalArgumentException(
210                 String.format("'%s' is not defined: %s", key, this)
211             );
212         }
213     }
214 
215     /**
216      * Convert entries to map.
217      * @param entries Entries
218      * @return Map
219      */
220     private static Map<String, String> fromEntries(final String... entries) {
221         final Map<String, String> result;
222         final int length = entries.length;
223         if (length == 0) {
224             result = new LinkedHashMap<>(0);
225         } else {
226             if (length % 2 != 0) {
227                 throw new IllegalArgumentException("Entries must be even");
228             }
229             final Map<String, String> res = new LinkedHashMap<>(0);
230             for (int idx = 0; idx < length; idx += 2) {
231                 res.put(entries[idx], entries[idx + 1]);
232             }
233             result = res;
234         }
235         return result;
236     }
237 
238     /**
239      * Parse attributes from XMIR.
240      * In the new method attributes are placed as a first data element.
241      * @param node Xmir node attribute.
242      * @return Map of attributes.
243      */
244     private static String fromXmir(final XmlNode node) {
245         final String original = node.text();
246         try {
247             return Attributes.NEW_LINE.matcher(
248                 new HexString(original.trim()).decode()
249             ).replaceAll("");
250         } catch (final NumberFormatException exception) {
251             throw new IllegalArgumentException(
252                 String.format(
253                     "Can't parse attributes '%s' from '%s' node, bytes %s",
254                     original,
255                     node,
256                     Arrays.toString(original.getBytes(StandardCharsets.UTF_8))
257                 ),
258                 exception
259             );
260         }
261     }
262 
263     /**
264      * Parse raw attributes.
265      * @param raw Raw attributes
266      * @return Map
267      */
268     private static Map<String, String> parse(final String raw) {
269         try {
270             return Arrays.stream(raw.split("\\|")).map(entry -> entry.split("="))
271                 .filter(arr -> arr.length > 1)
272                 .peek(Attributes::entrySize)
273                 .map(entry -> new MapEntry<>(entry[0], entry[1]))
274                 .collect(Collectors.toMap(MapEntry::getKey, MapEntry::getValue));
275         } catch (final IllegalArgumentException exception) {
276             throw new IllegalArgumentException(
277                 String.format("Can't parse attributes '%s'", raw),
278                 exception
279             );
280         }
281     }
282 
283     /**
284      * Check entry size.
285      * @param entry Entry to check
286      */
287     private static void entrySize(final String... entry) {
288         if (entry.length != 2) {
289             throw new IllegalArgumentException(
290                 String.format(
291                     "Entry must have two parts, but it has %d: %s",
292                     entry.length,
293                     Arrays.toString(entry)
294                 )
295             );
296         }
297     }

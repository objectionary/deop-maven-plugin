58  /**
59   * High-level representation of Opeo nodes.
60   * @since 0.1
61   */
62  @SuppressWarnings("PMD.AvoidDuplicateLiterals")
63  final class XmirParser {
64  
65      /**
66       * Opeo nodes.
67       */
68      private final List<XmlNode> nodes;
69  
70      /**
71       * Constructor.
72       * @param nodes Opeo nodes.
73       */
74      XmirParser(final AstNode... nodes) {
75          this(
76              Arrays.stream(nodes)
77                  .map(AstNode::toXmir)
78                  .map(Xembler::new)
79                  .map(Xembler::xmlQuietly)
80                  .map(XmlNode::new)
81                  .collect(Collectors.toList())
82          );
83      }
84  
85      /**
86       * Constructor.
87       * @param nodes Opeo nodes.
88       */
89      XmirParser(final List<XmlNode> nodes) {
90          this.nodes = nodes;
91      }
92  
93      /**
94       * Convert to XML nodes.
95       * @return XML nodes.
96       */
97      List<XmlNode> toJeoNodes() {
98          return this.nodes.stream()
99              .map(this::opcodes)
100             .flatMap(List::stream)
101             .collect(Collectors.toList());
102     }
103 
104     /**
105      * Convert XmlNode into a list of opcodes.
106      * @param node XmlNode
107      * @return List of opcodes
108      */
109     private List<XmlNode> opcodes(final XmlNode node) {
110         return this.node(node).opcodes()
111             .stream()
112             .map(AstNode::toXmir)
113             .map(Xembler::new)
114             .map(Xembler::xmlQuietly)
115             .map(XmlNode::new)
116             .collect(Collectors.toList());
117     }
118 
119     /**
120      * Convert XmlNode to AstNode.
121      * @param node XmlNode
122      * @return Ast node
123      * @checkstyle CyclomaticComplexityCheck (200 lines)
124      * @checkstyle ExecutableStatementCountCheck (200 lines)
125      * @checkstyle JavaNCSSCheck (200 lines)
126      * @checkstyle NestedIfDepthCheck (200 lines)
127      * @checkstyle MethodLengthCheck (200 lines)
128      * @todo #77:90min Refactor this.node() method.
129      *  The parsing method this.node() looks overcomplicated and violates many
130      *  code quality standards. We should refactor the method and remove all
131      *  the checkstyle and PMD "suppressions" from the method.
132      * @todo #110:90min Remove ad-hoc solution for replacing descriptors and owners.
133      *  Currently we have an ad-hoc solution for replacing descriptors and owners.
134      *  It looks ugly and requires refactoring. To remove ad-hoc solution we need
135      *  to remove all the if statements that use concerete class names as:
136      *  - "org/eolang/benchmark/B"
137      *  - "org/eolang/benchmark/BA"
138      */
139     @SuppressWarnings({"PMD.NcssCount", "PMD.ExcessiveMethodLength"})
140     private AstNode node(final XmlNode node) {
141         final AstNode result;
142         final String base = node.attribute("base").orElseThrow(
143             () -> new IllegalArgumentException(
144                 String.format(
145                     "Can't recognize node: %n%s%n'base' attribute should be present",
146                     node
147                 )
148             )
149         );
150         if (".plus".equals(base)) {
151             result = new Add(node, this::node);
152         } else if (".minus".equals(base)) {
153             result = new Substraction(node, this::node);
154         } else if ("cast".equals(base)) {
155             result = new Cast(node, this::node);
156         } else if ("opcode".equals(base)) {
157             final XmlInstruction instruction = new XmlInstruction(node.node());
158             final int opcode = instruction.opcode();
159             result = new Opcode(opcode, instruction.operands());
160         } else if ("label".equals(base)) {
161             result = new Label(this.node(node.children().collect(Collectors.toList()).get(0)));
162         } else if ("int".equals(base)) {
163             result = new Literal(node);
164         } else if ("string".equals(base)) {
165             result = new Literal(node);
166         } else if ("long".equals(base)) {
167             result = new Literal(node);
168         } else if (".super".equals(base)) {
169             final List<XmlNode> inner = node.children().collect(Collectors.toList());
170             final AstNode instance = this.node(inner.get(0));
171             result = new Super(
172                 instance,
173                 this.args(inner),
174                 node.attribute("scope")
175                     .orElseThrow(
176                         () -> new IllegalArgumentException(
177                             "Can't find descriptor for super invocation"
178                         )
179                     )
180             );
181         } else if ("$".equals(base)) {
182             result = new This(node);
183         } else if ("staticfield".equals(base)) {
184             result = new ClassField(new Attributes(node.attribute("scope").orElseThrow()));
185         } else if (".writearray".equals(base)) {
186             final List<XmlNode> inner = node.children().collect(Collectors.toList());
187             final AstNode array = this.node(inner.get(0));
188             final AstNode index = this.node(inner.get(1));
189             final AstNode value = this.node(inner.get(2));
190             result = new StoreArray(array, index, value);
191         } else if (".write".equals(base)) {
192             final List<XmlNode> inner = node.children().collect(Collectors.toList());
193             final AstNode target = this.node(inner.get(0));
194             final AstNode value = this.node(inner.get(1));
195             result = new VariableAssignment((LocalVariable) target, value);
196         } else if (".getfield".equals(base)) {
197             final List<XmlNode> inner = node.children().collect(Collectors.toList());
198             final XmlNode field = inner.get(0);
199             result = new FieldRetrieval(
200                 new Field(
201                     this.node(field.children().collect(Collectors.toList()).get(0)),
202                     new Attributes(field.attribute("scope").orElseThrow())
203                 )
204             );
205         } else if (".writefield".equals(base)) {
206             final List<XmlNode> inner = node.children().collect(Collectors.toList());
207             final XmlNode field = inner.get(0);
208             final AstNode value = this.node(inner.get(1));
209             result = new FieldAssignment(
210                 new Field(
211                     this.node(field.children().collect(Collectors.toList()).get(0)),
212                     new Attributes(field.attribute("scope").orElseThrow())
213                 ),
214                 value
215             );
216         } else if (base.contains("local")) {
217             result = new LocalVariable(node);
218         } else if (".new".equals(base)) {
219             final List<XmlNode> inner = node.children().collect(Collectors.toList());
220             final String type = inner.get(0).attribute("base").orElseThrow(
221                 () -> new IllegalArgumentException(
222                     String.format(
223                         "Can't find type of '%s'",
224                         base
225                     )
226                 )
227             );
228             final List<AstNode> args = this.args(inner);
229             final Attributes attributes = new Attributes()
230                 .descriptor(new ConstructorDescriptor(args).toString());
231             result = new Constructor(type, attributes, args);
232         } else if (".array".equals(base)) {
233             final List<XmlNode> children = node.children().collect(Collectors.toList());
234             final String type = new HexString(children.get(0).text()).decode();
235             final AstNode size = this.node(children.get(1));
236             result = new ArrayConstructor(size, type);
237         } else if (!base.isEmpty() && base.charAt(0) == '.') {
238             final Attributes attributes = new Attributes(
239                 node.attribute("scope")
240                     .orElseThrow(
241                         () -> new IllegalArgumentException(
242                             String.format(
243                                 "Can't find attributes of '%s'",
244                                 base
245                             )
246                         )
247                     )
248             );
249             if ("static".equals(attributes.type())) {
250                 result = new StaticInvocation(
251                     node, this.args(node.children().collect(Collectors.toList()))
252                 );
253             } else if ("interface".equals(attributes.type())) {
254                 result = new InterfaceInvocation(node, this::node);
255             } else {
256                 final List<XmlNode> inner = node.children().collect(Collectors.toList());
257                 final AstNode target = this.node(inner.get(0));
258                 result = new Invocation(target, attributes, this.args(inner));
259             }
260         } else {
261             throw new IllegalArgumentException(
262                 String.format("Can't recognize node: %n%s%n", node)
263             );
264         }
265         return result;
266     }
267 
268     /**
269      * Convert XML nodes into a list of arguments.
270      * @param all XML nodes.
271      * @return List of arguments.
272      */
273     private List<AstNode> args(final List<XmlNode> all) {
274         final List<AstNode> arguments;
275         if (all.size() > 1) {
276             arguments = all.subList(1, all.size())
277                 .stream()
278                 .map(this::node)
279                 .collect(Collectors.toList());
280         } else {
281             arguments = Collections.emptyList();
282         }
283         return arguments;
284     }

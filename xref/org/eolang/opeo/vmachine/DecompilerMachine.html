45  /**
46   * Decompiler machine.
47   * @since 0.1
48   */
49  public final class DecompilerMachine {
50  
51      /**
52       * Operand Stack.
53       */
54      private final Deque<Object> stack;
55  
56      /**
57       * Output root.
58       */
59      private final Root root;
60  
61      /**
62       * Instruction handlers.
63       */
64      private final Map<Integer, InstructionHandler> handlers;
65  
66      /**
67       * Arguments provided to decompiler.
68       */
69      private final Map<String, String> arguments;
70  
71      /**
72       * Constructor.
73       */
74      public DecompilerMachine() {
75          this(new HashMap<>());
76      }
77  
78      /**
79       * Constructor.
80       * @param args Arguments provided to decompiler.
81       */
82      public DecompilerMachine(final Map<String, String> args) {
83          this(new LinkedList<>(), args);
84      }
85  
86      /**
87       * Constructor.
88       * @param stack Operand stack.
89       * @param args Arguments provided to decompiler.
90       */
91      private DecompilerMachine(final Deque<Object> stack, final Map<String, String> args) {
92          this.stack = stack;
93          this.root = new Root();
94          this.arguments = args;
95          this.handlers = new MapOf<>(
96              new MapEntry<>(Opcodes.ICONST_1, new IconstHandler()),
97              new MapEntry<>(Opcodes.ICONST_2, new IconstHandler()),
98              new MapEntry<>(Opcodes.ICONST_3, new IconstHandler()),
99              new MapEntry<>(Opcodes.ICONST_4, new IconstHandler()),
100             new MapEntry<>(Opcodes.ICONST_5, new IconstHandler()),
101             new MapEntry<>(Opcodes.ALOAD, new AloadHandler()),
102             new MapEntry<>(Opcodes.NEW, new NewHandler()),
103             new MapEntry<>(Opcodes.DUP, new DupHandler()),
104             new MapEntry<>(Opcodes.BIPUSH, new BipushHandler()),
105             new MapEntry<>(Opcodes.INVOKESPECIAL, new InvokespecialHandler()),
106             new MapEntry<>(Opcodes.INVOKEVIRTUAL, new InvokevirtualHandler()),
107             new MapEntry<>(Opcodes.LDC, new LdcHandler()),
108             new MapEntry<>(Opcodes.POP, new PopHandler()),
109             new MapEntry<>(Opcodes.RETURN, new ReturnHandler())
110         );
111     }
112 
113     /**
114      * Decompile instructions into directives.
115      * @param instructions Instructions to decompile.
116      * @return Decompiled instructions.
117      */
118     public Iterable<Directive> decompileToXmir(final Instruction... instructions) {
119         Arrays.stream(instructions)
120             .forEach(inst -> this.handler(inst.opcode()).handle(inst));
121         return this.root.toXmir();
122     }
123 
124     /**
125      * Decompile instructions.
126      * @param instructions Instructions to decompile.
127      * @return Decompiled code.
128      */
129     public String decompile(final Instruction... instructions) {
130         Arrays.stream(instructions)
131             .forEach(inst -> this.handler(inst.opcode()).handle(inst));
132         return this.root.print();
133     }
134 
135     /**
136      * Do we add number to opcode name or not?
137      * if true then we add number to opcode name:
138      *  RETURN -> RETURN-1
139      * if false then we do not add number to opcode name:
140      *  RETURN -> RETURN
141      * @return Opcodes counting.
142      */
143     private boolean counting() {
144         return this.arguments.getOrDefault("counting", "true").equals("true");
145     }
146 
147     /**
148      * Get instruction handler.
149      * @param opcode Instruction opcode.
150      * @return Instruction handler.
151      */
152     private InstructionHandler handler(final int opcode) {
153         return this.handlers.getOrDefault(opcode, new UnimplementedHandler());
154     }
155 
156     /**
157      * Pops n arguments from the stack.
158      * @param number Number of arguments to pop.
159      * @return List of arguments.
160      */
161     private List<AstNode> popArguments(final int number) {
162         final List<AstNode> args = new LinkedList<>();
163         for (int index = 0; index < number; ++index) {
164             final Object arg = this.stack.pop();
165             final AstNode node = this.root.child(String.valueOf(arg))
166                 .orElseGet(() -> new Literal(arg));
167             this.root.disconnect(node);
168             args.add(node);
169         }
170         return args;
171     }

173     /**
174      * Instruction handler.
175      * @since 0.1
176      */
177     private interface InstructionHandler {
178 
179         /**
180          * Handle instruction.
181          * @param instruction Instruction to handle.
182          */
183         void handle(Instruction instruction);
184 
185     }

187     /**
188      * Aload instruction handler.
189      * @since 0.1
190      */
191     private class AloadHandler implements InstructionHandler {
192 
193         @Override
194         public void handle(final Instruction instruction) {
195             if (instruction.operand(0).equals(0)) {
196                 DecompilerMachine.this.stack.push("this");
197             }
198             DecompilerMachine.this.root.append(
199                 new Opcode(
200                     instruction.opcode(),
201                     instruction.operands(),
202                     DecompilerMachine.this.counting()
203                 )
204             );
205         }
206 
207     }
208

210      * New instruction handler.
211      * @since 0.1

213     private class NewHandler implements InstructionHandler {
214         @Override
215         public void handle(final Instruction instruction) {
216             DecompilerMachine.this.stack.push(
217                 new ObjectReference((String) instruction.operand(0)).toString()
218             );
219         }
220 
221     }

279     /**
280      * Invokespecial instruction handler.
281      * @since 0.1
282      */
283     private class InvokespecialHandler implements InstructionHandler {
284         @Override
285         public void handle(final Instruction instruction) {
286             if (!instruction.operand(1).equals("<init>")) {
287                 throw new UnsupportedOperationException(
288                     String.format("Instruction %s is not supported yet", instruction)
289                 );
290             }
291             if (instruction.operand(0).equals("java/lang/Object")) {
292                 DecompilerMachine.this.root.append(
293                     new Opcode(
294                         instruction.opcode(),
295                         instruction.operands(),
296                         DecompilerMachine.this.counting()
297                     )
298                 );
299             } else {
300                 final List<AstNode> args = DecompilerMachine.this.popArguments(
301                     Type.getArgumentCount((String) instruction.operand(2))
302                 );
303                 DecompilerMachine.this.root.append(
304                     new Constructor(
305                         (String) instruction.operand(0),
306                         (String) DecompilerMachine.this.stack.pop(),
307                         args
308                     )
309                 );
310             }
311         }
312 
313     }
314 
315     /**
316      * Invokevirtual instruction handler.
317      * @since 0.1
318      */
319     private class InvokevirtualHandler implements InstructionHandler {
320         @Override
321         public void handle(final Instruction instruction) {
322             final String method = (String) instruction.operand(1);
323             final String descriptor = (String) instruction.operand(2);
324             final List<AstNode> args = DecompilerMachine.this.popArguments(
325                 Type.getArgumentCount(descriptor)
326             );
327             final AstNode source = DecompilerMachine.this.popArguments(1).get(0);
328             DecompilerMachine.this.root.append(
329                 new Invocation(source, method, args)
330             );
331         }
332 
333     }

365     /**
366      * Iconst instruction handler.
367      * @since 0.1
368      */
369     private class IconstHandler implements InstructionHandler {
370 
371         @Override
372         public void handle(final Instruction instruction) {
373             switch (instruction.opcode()) {
374                 case Opcodes.ICONST_1:
375                     DecompilerMachine.this.stack.push(1);
376                     break;
377                 case Opcodes.ICONST_2:
378                     DecompilerMachine.this.stack.push(2);
379                     break;
380                 case Opcodes.ICONST_3:
381                     DecompilerMachine.this.stack.push(3);
382                     break;
383                 case Opcodes.ICONST_4:
384                     DecompilerMachine.this.stack.push(4);
385                     break;
386                 case Opcodes.ICONST_5:
387                     DecompilerMachine.this.stack.push(5);
388                     break;
389                 default:
390                     throw new UnsupportedOperationException(
391                         String.format("Instruction %s is not supported yet", instruction)
392                     );
393             }
394         }
